14   # Convert an enumerable to a hash.
 
15   #
 
16   # ==Example
 
17   #   array=[[:a, :b],[:c, :d],[:e, :f]]
 
18   #   array.to_h => {:a=>:b, :c=>:d, :e=>:f}
 
19   #
 
20   # If a key occurs more than once, then this will automatically 
21   # convert the value to an array of the keys' values.
 
22   #
 
23   # ==Example
 
24   #   array=[[:a,:b],[:a,:c],[:a,:d]]
 
25   #   array.to_h => {:a=>[:b, :c, :d]}
 
26 
27   def to_h
 
28     hash={}
 
29     dupe={}
 
30     each{|key,val|
 
31       if hash.key? key
 
32         if dupe.key? key
 
33           hash[key] << val
 
34         else
 
35           hash[key]=[hash[key]]
 
36           hash[key] << val
 
37           dupe[key]=true
 
38         end
 
39       else
 
40         hash[key]=val
 
41       end
 
42     }
 
43     return hash
 
44   end

326   #
 
327   # @nb This implementation uses #to_a and array intersection.

336   def intersect?(enum)
 
337     return ((self===enum and self.to_a.size>0) or ((self.to_a & enum.to_a).size>0))

341   # Return the cartesian product of the enumerations.
 
342   # http://en.wikipedia.org/wiki/Cartesian_product
 
343   #
 
344   # This is the fastest implementation we have found.
 
345   # It returns results in typical order.
 
346   #
 
347   # By Thomas Hafner
 
348   # See http://www.ruby-forum.com/topic/95519
 
349   #
 
350   # For our benchmarks, we also compared thesk:
 
351   # - By William James, http://www.ruby-forum.com/topic/95519
 
352   # - By Brian Schrer, http://blade.nagaokaut.ac.jp/cgi-bin/scat.rb/ruby/ruby-talk/151857
 
353 
354   def self.cartesian_product(*enums)
 
355     result = [[]]
 
356     while [] != enums
 
357       t, result = result, []
 
358       b, *enums = enums
 
359       t.each do |a|
 
360         b.each do |n|
 
361           result << a + [n]
 
362         end
 
363       end
 
364     end
 
365     result
 
366   end
