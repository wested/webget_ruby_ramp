16   #
 
17   # ==Typical Array#join with infix
 
18   #   list=['a','b','c']
 
19   #   list.join("*") => "a*b*c"
 
20   #
 
21   # ==Improved join with infix, prefix, suffix
 
22   #   list=['a','b','c']
 
23   #   list.join("*","[","]") => "[a]*[b]*[c]"
 
24   #
 
25   # ==Improved join with just prefix and suffix
 
26   #   list=['a','b','c']
 
27   #   list.join("[","]") => "[a][b][c]"

29   def join(*fixes)
 
30     if fixes.is_a?(String) then return self.ruby_join(fixes) end
 
31     case fixes.size
 
32     when 0
 
33       return self.ruby_join()
 
34     when 1
 
35       return self.ruby_join(fixes[0])
 
36     when 2
 
37       prefix=fixes[0].to_s
 
38       suffix=fixes[1].to_s
 
39       return self.map{|item| prefix + item.to_s + suffix}.ruby_join()
 
40     when 3
 
41       infix =fixes[0].to_s
 
42       prefix=fixes[1].to_s
 
43       suffix=fixes[2].to_s
 
44       return self.map{|item| prefix + item.to_s + suffix}.ruby_join(infix)
 
45     else
 
46       raise ArgumentError, "join() takes 0-3 arguments; you gave #{fixes.size}]"
 
47     end
 
48   end

154   # Divvy the array, like a pie, into _n_ number of slices.
 
155   #
 
156   # If the array divides evenly, then each slice has size/n items.
 
157   #
 
158   # Otherwise, divvy makes a best attempt by rounding up to give
 
159   # earlier slices one more item, which makes the last slice smaller:
 
160   #
 
161   # ==Examples
 
162   #   [1,2,3,4,5].divvy(2) => [[1,2,3],[4,5]]
 
163   #   [1,2,3,4,5,6,7].divvy(3) => [[1,2,3],[4,5,6],[7]]
 
164   #
 
165   # If the array size so small compared to _n_ that there is
 
166   # no mathematical way to _n_ slices, then divvy will return
 
167   # as many slices as it can.
 
168   #
 
169   # ==Examples
 
170   #   [1,2,3,4,5,6].divvy(4) => [[1,2],[3,4],[5,6]]
 
171 
172   def divvy(number_of_slices)

174     (number_of_slices > 0) or (raise ArgumentError, "number_of_slices must be > 0")
 
175     return slices((length.to_f/number_of_slices.to_f).ceil)
