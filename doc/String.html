prev_char(chr) click to toggle source

Return the previous character, with a changed flag and carry flag

Examples

  String.prev_char('n') => 'm', true, false   # change
  String.prev_char('a') => 'z', true, true    # change & carry
  String.prev_char('6') => '5', true, false   # change
  String.prev_char('0') => '9', true, true    # change & carry
  String.prev_char('-') => '-', false, false  # unchanged

     # File lib/webget_ruby_ramp/string.rb, line 109
109:  def self.prev_char(chr) #=> prev_char, changed_flag, carry_flag
110:   case chr
111:   when '1'..'9', 'B'..'Z', 'b'..'z'
112:     pos=(chr.respond_to?(:ord) ? chr.ord : chr[0])
113:     return (pos-1).chr, true, false
114:   when '0'
115:     return '9', true, true
116:   when 'A'
117:     return 'Z', true, true
118:   when 'a'
119:     return 'z', true, true
120:   else
121:     return chr, false, false
122:   end
