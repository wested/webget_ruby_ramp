prev_char(c) click to toggle source

Return the previous character, with a changed flag and carry flag

Examples

  String.prev_char('n') => 'm', true, false   # change
  String.prev_char('a') => 'z', true, true    # change & carry
  String.prev_char('6') => '5', true, false   # change
  String.prev_char('0') => '9', true, true    # change & carry
  String.prev_char('-') => '-', false, false  # unchanged

     # File lib/ramp/string.rb, line 120
120:  def self.prev_char(c) #=> prev_char, changed_flag, carry_flag
121:   case c
122:   when '1'..'9', 'B'..'Z', 'b'..'z'
123:     i=(c.respond_to?(:ord) ? c.ord : c[0])
124:     return (i-1).chr, true, false
125:   when '0'
126:     return '9', true, true
127:   when 'A'
128:     return 'Z', true, true
129:   when 'a'
130:     return 'z', true, true
131:   else
132:     return c, false, false
133:   end
