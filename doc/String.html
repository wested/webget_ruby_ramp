prev_char(c) click to toggle source

Return the previous character, with a changed flag and carry flag

Examples

  String.prev_char('n') => 'm', true, false   # change
  String.prev_char('a') => 'z', true, true    # change & carry
  String.prev_char('6') => '5', true, false   # change
  String.prev_char('0') => '9', true, true    # change & carry
  String.prev_char('-') => '-', false, false  # unchanged

    # File lib/ramp/string.rb, line 72
72:  def self.prev_char(c) #=> prev_char, changed_flag, carry_flag
73:   case c
74:   when '1'..'9', 'B'..'Z', 'b'..'z'
75:     i=(c.respond_to?(:ord) ? c.ord : c[0])
76:     return (i-1).chr, true, false
77:   when '0'
78:     return '9', true, true
79:   when 'A'
80:     return 'Z', true, true
81:   when 'a'
82:     return 'z', true, true
83:   else
84:     return c, false, false
85:   end
